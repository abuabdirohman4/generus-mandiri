# Documentation Site (`/docs`) Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Build a public, no-login documentation site at `generus.abuabdirohman.com/docs` using Fumadocs inside the existing Next.js 15 app.

**Architecture:** Fumadocs is integrated directly into the existing Next.js app router via `src/app/docs/` route. MDX content files live in `content/docs/`. No separate app, no monorepo. Robots meta tag set to `noindex` so search engines cannot crawl it.

**Tech Stack:** Fumadocs Core + Fumadocs UI, Next.js 15 App Router, MDX, Tailwind CSS 4.

## Task 3: Create Docs Layout and Routes

**Files:**
- Create: `src/app/docs/layout.tsx`
- Create: `src/app/docs/[[...slug]]/page.tsx`

**Step 1: Create `src/app/docs/layout.tsx`**

```typescript
import { DocsLayout } from 'fumadocs-ui/layouts/docs'
import { RootProvider } from 'fumadocs-ui/provider'
import { docsSource } from '@/lib/docs-source'
import 'fumadocs-ui/style.css'
import type { ReactNode } from 'react'

export default function Layout({ children }: { children: ReactNode }) {
  return (
    <RootProvider>
      <DocsLayout tree={docsSource.pageTree} nav={{ title: 'Generus Mandiri Docs' }}>
        {children}
      </DocsLayout>
    </RootProvider>
  )
}
```

**Step 2: Create `src/app/docs/[[...slug]]/page.tsx`**

```typescript
import { docsSource } from '@/lib/docs-source'
import { DocsPage, DocsBody, DocsTitle, DocsDescription } from 'fumadocs-ui/page'
import { MDXContent } from 'fumadocs-mdx'
import { notFound } from 'next/navigation'

interface Props {
  params: Promise<{ slug?: string[] }>
}

export default async function Page({ params }: Props) {
  const { slug } = await params
  const page = docsSource.getPage(slug)
  if (!page) notFound()

  return (
    <DocsPage toc={page.data.toc}>
      <DocsTitle>{page.data.title}</DocsTitle>
      <DocsDescription>{page.data.description}</DocsDescription>
      <DocsBody>
        <MDXContent code={page.data.body} />
      </DocsBody>
    </DocsPage>
  )
}

export function generateStaticParams() {
  return docsSource.generateParams()
}

export async function generateMetadata({ params }: Props) {
  const { slug } = await params
  const page = docsSource.getPage(slug)
  if (!page) notFound()

  return {
    title: page.data.title,
    description: page.data.description,
    robots: { index: false, follow: false },
  }
}
```

**Step 3: Commit**

```bash
git add src/app/docs/
git commit -m "feat(docs): add fumadocs layout and dynamic page route"
```
